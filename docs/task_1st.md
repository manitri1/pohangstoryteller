## 📌 데이터 구조

### 🏗️ 코스 데이터 모델

```typescript
interface Course {
  id: string;
  title: string;
  description: string;
  category: '맛집탐방' | '역사여행' | '자연경관' | '골목산책';
  duration: number; // 분 단위
  difficulty: '쉬움' | '보통' | '어려움';
  locations: Location[];
  media: Media[];
}

interface Location {
  id: string;
  name: string;
  description: string;
  coordinates: { lat: number; lng: number };
  qrCode: string;
  media: Media[];
}

interface Media {
  id: string;
  type: 'image' | 'video' | 'text';
  url: string;
  title: string;
  duration?: number; // 영상의 경우
}
```
